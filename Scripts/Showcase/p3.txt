local soundService = game:GetService("SoundService")
local tick = soundService:WaitForChild("fc")
local ffdActivation = soundService:WaitForChild("ffd")
local notification = require(game.ReplicatedStorage.Notification).sg
local indicator = workspace:WaitForChild("Buttonu1"):WaitForChild("Indicator"):WaitForChild("Glass")
local screen = workspace:WaitForChild("Screen")
local statusLabel = screen:WaitForChild("SurfaceGui"):WaitForChild("TextLabel")
local descriptionLabel
for _, v in ipairs(screen.SurfaceGui:GetChildren()) do
    if v:IsA("TextLabel") and v ~= statusLabel and v:FindFirstChild("UIStroke") then
        descriptionLabel = v
        break
    end
end
game.ReplicatedStorage.p3.OnClientEvent:Connect(function()
    indicator.Material = Enum.Material.Neon
    indicator.Color = Color3.fromRGB(0, 255, 0)
    task.delay(1.3, function()
        statusLabel.Text = "<< Start Engaged >>"
        descriptionLabel.Text = "<< Protocol T-SD-02 activated >>"
        statusLabel.TextSize = 64
        coroutine.wrap(function()
            task.wait(0.4)
            for v = 1, 4 do
                statusLabel.Text = ""
                task.wait(0.4)
                statusLabel.Text = "<< Start Engaged >>"
                task.wait(0.4)
            end
        end)()
    end)
    ffdActivation:Play()
    task.delay(6, function()
        local txt = "Attention all facility personnel! Protocol T-SD-02 has been\nactivated. Full Facility Destruction will be ready for engaging in\nT - 5 minutes! Please evacuate, immediately!"
        notification(txt, 13)
    end)
    task.wait(20.65)
    ffdActivation.Ended:Wait()
    task.wait(0.1)
    statusLabel.Text = "<< Preparing FFD for engaging.. >>"
    statusLabel.TextSize = 44
    for v = 19, 0, -1 do
        tick:Play()
        descriptionLabel.Text = "<< FFD will be ready for engaging in T - " .. v .. "s >>"
        task.wait(1)
    end
    statusLabel.Text = "<< FFD is now ready for engaging >>"
    descriptionLabel.Text = "<< Awaiting confirmation signal >>"
end)
