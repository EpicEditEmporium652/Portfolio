local prefix = ";"
local data = game:GetService("DataStoreService")
local a = data:GetDataStore("Admins")
local b = data:GetDataStore("Banned")
 
local d
local sc, er = pcall(function()
    d = a:GetAsync("Users") or {}
end)
if not sc then rmt:FireClient(p, "nf", er) d = d or {} end
local admins = {}
admins[game.Players:GetNameFromUserIdAsync(game.CreatorId)] = true
task.spawn(function()
    if not d then return end
    for _, v in ipairs(d) do
        admins[v] = true
    end
end)
 
local bn
local sc, er = pcall(function()
    bn = b:GetAsync("Users") or {}
end)
if not sc then rmt:FireClient(p, "nf", er) bn = bn or {} end
 
local function gp(p, d)
    local mt = {}
    local p = p:lower()
    if p == "me" then
        return d
    elseif p == "others" then
        for _, v in ipairs(game.Players:GetPlayers()) do
            if v ~= d then table.insert(mt, v) end
        end
        return mt
    elseif p == "all" then
        for _, v in ipairs(game.Players:GetPlayers()) do
            table.insert(mt, v)
        end
        return mt
    end
    for _, v in ipairs(game.Players:GetPlayers()) do
        if (v.Name:sub(1, #p):lower() == p or v.DisplayName:sub(1, #p):lower() == p) then
            table.insert(mt, v)
        end
    end
    if #mt == 1 then
        return mt[1]
    else
        return nil
    end
end
 
game.Players.PlayerAdded:Connect(function(p)
    task.spawn(function()
        if not bn then return end
        for _, v in ipairs(bn) do
            if p.Name == v then
                p:Kick("You're Banned")
            end
        end
    end)
    p.Chatted:Connect(function(m)
        if not admins[p.Name] then return end
        local arg, arg2, arg3
        local rmt = game.ReplicatedStorage.f
        local plr
        if m:lower():find(" ") then
            local splt = m:split(" ")
            arg, arg2 = splt[1], splt[2]
            if splt[3] then arg3 = splt[3] end
            plr = gp(arg2, p)
        else
            arg = m
            plr = p
        end
        
        if arg:lower() == prefix .. "fly" then
            if plr then
                if typeof(plr) == "table" then
                    for _, v in ipairs(plr) do
                        rmt:FireClient(v, "fly")
                    end
                    rmt:FireClient(p, "nf", "Success")
                else
                    rmt:FireClient(plr, "fly")
                    rmt:FireClient(p, "nf", "Success")
                end
            else
            rmt:FireClient(p, "nf", "Error")
            end
            
        elseif arg == prefix .. "unfly" then
            if plr then
                if typeof(plr) == "table" then
                    for _, v in ipairs(plr) do
                        rmt:FireClient(v, "un")
                    end
                    rmt:FireClient(p, "nf", "Success")
                else
                    rmt:FireClient(plr, "un")
                    rmt:FireClient(p, "nf", "Success")
                end
            else
            rmt:FireClient(p, "nf", "Error")
            end
            
        elseif arg == prefix .. "kick" then
            if plr then
                if typeof(plr) == "table" then
                    for _, v in ipairs(plr) do
                        if admins[v.Name] then return end
                        v:Kick((arg3 and ("\n" .. arg3)) or ("\nKicked by " .. p.Name))
                    end
                    rmt:FireClient(p, "nf", "Success")
                else
                    if admins[plr.Name] then return end
                    plr:Kick((arg3 and ("\n" .. arg3)) or ("\nKicked by " .. p.Name))
                    rmt:FireClient(p, "nf", "Success")
                end
            else
            rmt:FireClient(p, "nf", "Error")
            end
            
        elseif arg == prefix .. "ban" then
            if plr then
                if typeof(plr) == "table" then
                    for _, v in ipairs(plr) do
                        if admins[v.Name] then return end
                        table.insert(bn, v.Name)
                        local sc, er = pcall(function()
                            b:SetAsync("Users", bn)
                        end)
                        if not sc then rmt:FireClient(p, "nf", "Failed to store and ban : " .. v.Name .. " : " .. er) return end
                        v:Kick((arg3 and ("\n" .. arg3)) or ("\nBanned by " .. p.Name))
                    end
                    rmt:FireClient(p, "nf", "Success")
                else
                    if admins[plr.Name] then return end
                    table.insert(bn, plr.Name)
                    local sc, er = pcall(function()
                        b:SetAsync("Users", bn)
                    end)
                    if not sc then rmt:FireClient(p, "nf", "Failed to store and ban : " .. plr.Name .. " : " .. er) return end
                    plr:Kick((arg3 and ("\n" .. arg3)) or ("\nBanned by " .. p.Name))
                    rmt:FireClient(p, "nf", "Success")
                end
            else
            rmt:FireClient(p, "nf", "Error")
            end
            
        elseif arg == prefix .. "setrank" then
            if plr then
                if admins[plr.Name] then return end
                if typeof(plr) == "table" then
                    for _, v in ipairs(plr) do
                        admins[v.Name] = true
                    end
                    rmt:FireClient(p, "nf", "Success")
                else
                    admins[plr.Name] = true
                    rmt:FireClient(p, "nf", "Success")
                end
            else
            rmt:FireClient(p, "nf", "Error")
            end
            
        elseif arg == prefix .. "permrank" then
            if plr then
                if admins[plr.Name] then return end
                if not d then return end
                if typeof(plr) == "table" then
                    for _, v in ipairs(plr) do
                        table.insert(d, v.Name)
                        local sc, er = pcall(function()
                            a:SetAsync("Users", d)
                        end)
                        if not sc then rmt:FireClient(p, "nf", "Failed to store and rank : " .. v.Name .. " : " .. er) return end
                        admins[v.Name] = true
                    end
                    rmt:FireClient(p, "nf", "Success")
                else
                    table.insert(d, plr.Name)
                    local sc, er = pcall(function()
                        a:SetAsync("Users", d)
                    end)
                    if not sc then rmt:FireClient(p, "nf", "Failed to store and rank : " .. plr.Name .. " : " .. er) return end
                    admins[plr.Name] = true
                    rmt:FireClient(p, "nf", "Success")
                end
            else
            rmt:FireClient(p, "nf", "Error")
            end
            
        elseif arg == prefix .. "removerank" then
            if plr then
                if plr.UserId == game.CreatorId then return end
                if not d then return end
                if typeof(plr) == "table" then
                    for _, v in ipairs(plr) do
                        local i = table.find(d, v.Name)
                        if i then
                            table.remove(d, i)
                            admins[v.Name] = false
                        else
                            return
                        end
                        local sc, er = pcall(function()
                            a:SetAsync("Users", d)
                        end)
                        if not sc then rmt:FireClient(p, "nf", "Failed to store and removerank : " .. v.Name .. " : " .. er) return end
                    end
                    rmt:FireClient(p, "nf", "Success")
                else
                    local i = table.find(d, plr.Name)
                    if i then
                        table.remove(d, i)
                        admins[v.Name] = false
                    else
                        return
                    end
                    local sc, er = pcall(function()
                        a:SetAsync("Users", d)
                    end)
                    if not sc then rmt:FireClient(p, "nf", "Failed to store and removerank : " .. plr.Name .. " : " .. er) return end
                    rmt:FireClient(p, "nf", "Success")
                end
            else
            rmt:FireClient(p, "nf", "Error")
            end
            
        elseif arg == prefix .. "offrank" then
            if plr then return end
            if admins[arg2] then return end
            if not arg2 then return end
            if not d then return end
            local sc, er = pcall(function()
                game.Players:GetUserIdFromNameAsync(arg2)
            end)
            if not sc then rmt:FireClient(p, "nf", "Username Invalid : " .. arg2) return end
            admins[arg2] = true
            table.insert(d, arg2)
            local sc, er = pcall(function()
                a:SetAsync("Users", d)
            end)
            if not sc then rmt:FireClient(p, "nf", "Failed to rank : " .. arg2 .. " : " .. er) return end
            rmt:FireClient(p, "nf", "Error")
            
        elseif arg == prefix .. "removeoffrank" then
            if plr then return end
            if not arg2 then return end
            if not d then return end
            local i = table.find(d, arg2)
            if i then
                admins[arg2] = false
                table.remove(d, i)
            else
                return
            end
            local sc, er = pcall(function()
                a:SetAsync("Users", d)
            end)
            if not sc then rmt:FireClient(p, "nf", "Failed to removeoffrank : " .. arg2 .. " : " .. er) return end
            rmt:FireClient(p, "nf", "Success")
            
        elseif arg == prefix .. "rs" then
            local sc, er = pcall(function()
                game:GetService("TeleportService"):TeleportToPrivateServer(game.PlaceId, game:GetService("TeleportService"):ReserveServer(game.PlaceId), game.Players:GetPlayers())
            end)
            if not sc then rmt:FireClient(p, "nf", "Failed to restart server : " .. er) end
            
        elseif arg == prefix .. "goto" then
            if plr then
                if typeof(plr) == "table" then return end
                if p.Character and plr.Character and p.Character:FindFirstChild("HumanoidRootPart") and plr.Character:FindFirstChild("HumanoidRootPart") then
                    p.Character:MoveTo(plr.Character.HumanoidRootPart.Position)
                end
            else
            rmt:FireClient(p, "nf", "Error")
            end
            
        elseif arg == prefix .. "bring" then
            if plr then
                if typeof(plr) == "table" then
                    for _, v in ipairs(plr) do
                        if p.Character and v.Character and p.Character:FindFirstChild("HumanoidRootPart") and v.Character:FindFirstChild("HumanoidRootPart") then
                            v.Character:MoveTo(p.Character.HumanoidRootPart.Position)
                        end
                    end
                else
                    if p.Character and plr.Character and p.Character:FindFirstChild("HumanoidRootPart") and plr.Character:FindFirstChild("HumanoidRootPart") then
                        plr.Character:MoveTo(p.Character.HumanoidRootPart.Position)
                    end
                end
            else
            rmt:FireClient(p, "nf", "Error")
            end
            
        elseif arg == prefix .. "cmds" then
            if p.PlayerGui:FindFirstChild("cmd") then return end
            if arg2 then return end
            local cmd = game.ServerStorage.Screengui:Clone()
            cmd.Name = "cmd"
            cmd.Parent = p.PlayerGui
            cmd.Frame.kick.TextButton.MouseButton1Click:Connect(function()
                cmd:Destroy()
            end)                
            
        end
    end)
end)
